        (function() {
            // ==================== 星空背景 ====================

                const star = document.createElement('div');

                star.style.width = size + 'px';

                starsContainer.appendChild(star);
            }

            // ==================== 烟花Canvas系统 ====================
            const canvas = document.getElementById('fireworkCanvas');
            const ctx = canvas.getContext('2d');
            let canvasW, canvasH;

            function resizeCanvas() {
                canvasW = window.innerWidth;
                canvasH = window.innerHeight;
                canvas.width = canvasW;
                canvas.height = canvasH;
            }
            resizeCanvas();
            window.addEventListener('resize', resizeCanvas);
            window.addEventListener('orientationchange', () => {
                setTimeout(resizeCanvas, 200);

            // 粒子类
            class Particle {
                constructor(x, y, color, velocity, life, size) {
                    this.x = x;
                    this.y = y;
                    this.color = color;
                    this.velocity = velocity;
                    this.life = life;
                    this.maxLife = life;
                    this.size = size || (Math.random() * 2.5 + 1.5);
                    this.gravity = 0.04;
                    this.friction = 0.985;
                    this.alpha = 1;
                }
                update() {
                    this.velocity.x *= this.friction;
                    this.velocity.y *= this.friction;
                    this.velocity.y += this.gravity;
                    this.x += this.velocity.x;
                    this.y += this.velocity.y;
                    this.life--;
                    this.alpha = this.life / this.maxLife;
                }
                draw(ctx) {
                    ctx.save();
                    ctx.globalAlpha = this.alpha;
                    ctx.fillStyle = this.color;
                    ctx.shadowColor = this.color;
                    ctx.shadowBlur = this.size * 3;
                    ctx.beginPath();
                    ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                    ctx.fill();
                    // 光晕
                    ctx.globalAlpha = this.alpha * 0.4;
                    ctx.shadowBlur = this.size * 8;
                    ctx.beginPath();
                    ctx.arc(this.x, this.y, this.size * 1.8, 0, Math.PI * 2);
                    ctx.fill();
                    ctx.restore();
                }
                get isDead() { return this.life <= 0; }
            }

            // 烟花火箭类
            class Rocket {
                constructor(startX, startY, targetX, targetY) {
                    this.x = startX;
                    this.y = startY;
                    this.targetX = targetX;
                    this.targetY = targetY;
                    this.speed = Math.random() * 4 + 7;
                    const dx = targetX - startX;
                    const dy = targetY - startY;
                    const dist = Math.sqrt(dx * dx + dy * dy);
                    this.vx = (dx / dist) * this.speed;
                    this.vy = (dy / dist) * this.speed;
                    this.trail = [];
                    this.maxTrail = 18;
                    this.alive = true;
                    this.trailColor = `hsl(${Math.random()*30+20}, 100%, ${Math.random()*30+55}%)`;
                }
                update() {
                    this.trail.push({ x: this.x, y: this.y, life: 8 });
                    if (this.trail.length > this.maxTrail) this.trail.shift();
                    this.trail.forEach(t => t.life--);
                    this.x += this.vx;
                    this.y += this.vy;
                    const dx = this.targetX - this.x;
                    const dy = this.targetY - this.y;
                    if (Math.sqrt(dx * dx + dy * dy) < 12) {
                        this.alive = false;
                    }
                    if (this.y < -50 || this.y > canvasH + 50 || this.x < -50 || this.x > canvasW + 50) {
                        this.alive = false;
                    }
                }
                draw(ctx) {
                    // 拖尾
                    for (let i = 0; i < this.trail.length; i++) {
                        const t = this.trail[i];
                        const alpha = t.life / 8;
                        ctx.save();
                        ctx.globalAlpha = alpha * 0.7;
                        ctx.fillStyle = this.trailColor;
                        ctx.shadowColor = this.trailColor;
                        ctx.shadowBlur = 5;
                        ctx.beginPath();
                        ctx.arc(t.x, t.y, 2, 0, Math.PI * 2);
                        ctx.fill();
                        ctx.restore();
                    }
        // 火箭主体
                    ctx.save();
                    ctx.fillStyle = '#fff';
                    ctx.shadowColor = '#ffdd99';
                    ctx.shadowBlur = 10;
                    ctx.beginPath();
                    ctx.arc(this.x, this.y, 3.5, 0, Math.PI * 2);
                    ctx.fill();
                    ctx.fillStyle = this.trailColor;
                    ctx.shadowColor = this.trailColor;
                    ctx.shadowBlur = 14;
                    ctx.beginPath();
                    ctx.arc(this.x, this.y, 2.5, 0, Math.PI * 2);
                    ctx.fill();
                    ctx.restore();
                }
            }

            // 烟花爆炸
            class Firework {
                constructor(x, y) {
                    this.particles = [];
                    this.x = x;
                    this.y = y;
                    this.alive = true;
                    const particleCount = Math.floor(Math.random() * 60 + 80);
                    const hueBase = Math.random() * 360;
                    const colorScheme = Math.floor(Math.random() * 4);

                    for (let i = 0; i < particleCount; i++) {
                        let hue, sat, light;
                        if (colorScheme === 0) {
                            // 粉-红-金色系
                            hue = hueBase + Math.random() * 40 - 20;
                            sat = 85 + Math.random() * 15;
                            light = 45 + Math.random() * 35;
                        } else if (colorScheme === 1) {
                            // 紫-蓝-粉系
                            hue = 260 + Math.random() * 80;
                            sat = 70 + Math.random() * 30;
                            light = 50 + Math.random() * 30;
                        } else if (colorScheme === 2) {
                            // 金-橙-红系
                            hue = 15 + Math.random() * 50;
                            sat = 90 + Math.random() * 10;
                            light = 50 + Math.random() * 30;
                        } else {
                            // 多彩混合
                            hue = Math.random() * 360;
                            sat = 80 + Math.random() * 20;
                            light = 48 + Math.random() * 32;
                        }
                        const color = `hsl(${hue}, ${sat}%, ${light}%)`;
                        const angle = Math.random() * Math.PI * 2;
                        const speed = Math.random() * 8 + 3;
                        const velocity = {
                            x: Math.cos(angle) * speed * (0.7 + Math.random() * 0.6),
                            y: Math.sin(angle) * speed * (0.7 + Math.random() * 0.6),
                        };
                        const life = Math.floor(Math.random() * 35 + 45);
                        const size = Math.random() * 2.2 + 1.2;
                        this.particles.push(new Particle(x, y, color, velocity, life, size));
                    }
                    // 中心闪光粒子
                    for (let i = 0; i < 20; i++) {
                        const color = `hsl(${hueBase+Math.random()*30}, 100%, ${80+Math.random()*20}%)`;
                        const angle = Math.random() * Math.PI * 2;
                        const speed = Math.random() * 2 + 0.5;
                        const velocity = { x: Math.cos(angle) * speed, y: Math.sin(angle) * speed };
                        this.particles.push(new Particle(x, y, color, velocity, 20 + Math.random() * 25, 1 + Math
                        .random()));
                    }
                }
                update() {
                    this.particles.forEach(p => p.update());
                    this.particles = this.particles.filter(p => !p.isDead);
                    if (this.particles.length === 0) this.alive = false;
                }
                draw(ctx) {
                    this.particles.forEach(p => p.draw(ctx));
                }
            }

            // 全局状态
            let rockets = [];

            function launchRocket(targetX, targetY) {
                const startX = targetX + (Math.random() - 0.5) * canvasW * 0.4;
                const startY = canvasH + 30;
                const tX = targetX + (Math.random() - 0.5) * 60;
                const tY = targetY + (Math.random() - 0.5) * 40;
                rockets.push(new Rocket(startX, startY, Math.max(30, Math.min(canvasW - 30, tX)), Math.max(40, Math.min(
                    canvasH * 0.6, tY))));
            }

            function explodeAt(x, y) {
                fireworks.push(new Firework(x, y));
            }

            // 动画循环
            function animate() {
                ctx.clearRect(0, 0, canvasW, canvasH);
